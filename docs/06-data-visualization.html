Visualization function

Data wrangling

parameterFactoring = function(dataset, parameters, ...) {
  new_data = dataset
  for(param in parameters) {
    new_data = new_data |> 
      mutate(param = factor(get(param), ...))
  }
  return(new_data)
}
iris2 = parameterFactoring(iris, "Species", levels = c("setosa", "versicolor", "virginica"), 
                          labels = c("Setosa", "Versicolor", "Virginica"))
str(iris2)

Visualization function

baseFig = function(dataset, parameters, x_param, y_param, z_param = NULL, color_list, ...) {
  p = dataset |> 
        ggplot(aes(x = get(x_param), y = get(y_param))) +
        geom_violin(aes(color = get(x_param), trim = FALSE, show.legend = FALSE)) +
        scale_color_manual(values = color_list)
  return(p)
}
f2 = baseFig(iris2, "Species", "Sepal.Length", "Petal.Length", 
       color_list = c("setosa" = "#06d6a0", "versicolor" = "#118ab2", 
                                "virginica" = "#073b4c"))
f2

Visualization function

pointsOverlay = function(plot, z_param = NULL, ...) {
  if(!is.null(z_param)) {
    p = plot + geom_jitter(aes(fill = get(z_param)), ...) + 
      scale_fill_viridis(name = gsub("\\.", " ", z_param))
  } else {
    p = plot + geom_jitter(...)
  }
  return(p)
}
f3 = pointsOverlay(f2, z_param = "Petal.Length", shape = 21, size = 3, alpha = 0.8)

Visualization function

addStats = function(plot, dataset, comparisons) {
  minY = max(layer_scales(plot)$y$range$range)
  yPositions = rep(minY, 3) + c(0.3, 0.6, 0.9)
  p = plot + 
    geom_signif(comparisons = comparisons, 
                map_signif_level = TRUE, 
                y_position = yPositions, 
                tip_length = 0)
}

f4 = addStats(f3, iris2, comparisons = list(c("Setosa", "Versicolor"),
